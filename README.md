# Clone the repository to your local machine
git clone https://github.com/your-username/your-repository.git

# Navigate into the project directory
cd your-repository

# Pulls (fetches and merges) changes from the remote branch
git pull origin <your-branch-name>

# List all local branches (the current branch is highlighted)
git branch

# Create a new branch
git branch new-feature

# Switch to the new branch (older command)
git checkout new-feature

# Switch to the new branch (preferred modern command)
git switch new-feature
